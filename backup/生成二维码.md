Kanji_encode(str)
{
    tmp := StrSplit(str)
    for k,v in tmp
    {
        c := Asc(v)
        If ((c >= 0x0001) && (c <= 0x007F))
            out .= v
        Else if (c > 0x07FF)
        {
            out .= Chr(0xE0 | ((c >> 12) & 0x0F))
            out .= Chr(0x80 | ((c >> 6) & 0x3F))
            out .= Chr(0x80 | ((c >> 0) & 0x3F))
        }
        Else
        {
            out .= Chr(0xC0 | ((c >> 6) & 0x1F))
            out .= Chr(0x80 | ((c >> 0) & 0x3F))
        }
    }
    Return out
}
